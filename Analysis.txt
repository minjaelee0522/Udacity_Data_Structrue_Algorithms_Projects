import csv
with open('texts.csv', 'r') as f:
    reader = csv.reader(f)
    texts = list(reader)
    print("First record of texts, {0} texts {1} at time {2}".format(texts[0][1], texts[0][0], texts[0][2]))      -> O(1)


with open('calls.csv', 'r') as f:
    reader = csv.reader(f)
    calls = list(reader)
    print("Last record of calls, {0} calls {1} at time {2}, lasting {3} seconds".format(calls[-1][1], calls[-1][0], calls[-1][2], calls[-1][3]))   -> O(5213) -> O(1)
